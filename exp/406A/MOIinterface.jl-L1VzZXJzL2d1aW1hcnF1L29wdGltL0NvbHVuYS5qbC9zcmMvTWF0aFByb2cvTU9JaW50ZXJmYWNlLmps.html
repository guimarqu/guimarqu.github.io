190 function _getreducedcost(form::Formulation, optimizer::MoiOptimizer, var::Variable)
191 sign = getobjsense(form) == MinSense ? 1.0 : -1.0
192 inner = getinner(optimizer)
193 if MOI.get(inner, MOI.ResultCount()) < 1
194 @warn """
195 No dual solution stored in the optimizer of formulation. Cannot retrieve reduced costs.
196 Method returns nothing.
197 """
198 return nothing
199 end
200 if !iscuractive(form, var) || !isexplicit(form, var)
201 varname = getname(form, var)
202 @warn """
203 Cannot retrieve reduced cost of variable $varname because the variable must be active and explicit.
204 Method returns nothing.
205 """
206 return nothing
207 end
208 bounds_interval_idx = getbounds(getmoirecord(var))
209 dualval = MOI.get(inner, MOI.ConstraintDual(1), bounds_interval_idx)
210 return sign * dualval
211 end

215 function get_primal_solutions(form::F, optimizer::MoiOptimizer) where {F <: Formulation}
216 inner = getinner(optimizer)
217 nb_primal_sols = MOI.get(inner, MOI.ResultCount())
218 solutions = PrimalSolution{F}[]
219 for res_idx in 1:nb_primal_sols
220 if MOI.get(inner, MOI.PrimalStatus(res_idx)) != MOI.FEASIBLE_POINT
221 continue
222 end
223 solcost = 0.0 # TODO : constant in the objective function ?
224 solvars = Vector{VarId}()
225 solvals = Vector{Float64}()
226 for (id, var) in getvars(form)

228 moirec = getmoirecord(var)
229 moi_index = getindex(moirec)
230 kind = _getcolunakind(moirec)
231 val = MOI.get(inner, MOI.VariablePrimal(res_idx), moi_index)
232 solcost += val * getcurcost(form, id)
233 val = round(val, digits = Coluna._params_.tol_digits)
234 if abs(val) > Coluna._params_.tol
235 @logmsg LogLevel(-4) string("Var ", var.name , " = ", val)
236 push!(solvars, id)

238 end
239 end

245 function get_dual_solutions(form::F, optimizer::MoiOptimizer) where {F <: Formulation}
246 inner = getinner(optimizer)
247 nb_dual_sols = MOI.get(inner, MOI.ResultCount())
248 solutions = DualSolution{F}[]
249 for res_idx in 1:nb_dual_sols
250 if MOI.get(inner, MOI.DualStatus(res_idx)) != MOI.FEASIBLE_POINT
251 continue
252 end
253 solcost = 0.0 # TODO : constant in the objective function ?
254 solconstrs = Vector{ConstrId}()
255 solvals = Vector{Float64}()
256 # Get dual value of constraints

258 iscuractive(form, id) && isexplicit(form, id) || continue
259 moi_index = getindex(getmoirecord(constr))
260 val = MOI.get(inner, MOI.ConstraintDual(res_idx), moi_index)
261 solcost += val * getcurrhs(form, id)
262 val = round(val, digits = Coluna._params_.tol_digits)
263 if abs(val) > Coluna._params_.tol
264 @logmsg LogLevel(-4) string("Constr ", constr.name, " = ", val)
265 push!(solconstrs, id)
266 push!(solvals, val)
267 end
268 end
269 # Get reduced cost of variables
270 for (id, var) in getvars(form)
271 iscuractive(form, id) && isexplicit(form, id) || continue
272 moi_bounds_index = getbounds(getmoirecord(var))
273 basis_status = MOI.get(inner, MOI.ConstraintBasisStatus(res_idx), moi_bounds_index)
274 val = MOI.get(inner, MOI.ConstraintDual(res_idx), moi_bounds_index)
275 if basis_status == MOI.NONBASIC_AT_LOWER
276 solcost += val * getcurlb(form, id)
277 elseif basis_status == MOI.NONBASIC_AT_UPPER
278 solcost += val * getcurub(form, id)
279 end
280 # TODO : store reduced cost of the variable in the dual constraint ?
281 end
282
283 sense = getobjsense(form) == MaxSense ? -1.0 : 1.0
284 push!(solutions, DualSolution(form, solconstrs, solvals, sense * solcost, FEASIBLE_SOL))
285 end
286 return solutions
