35 """
36 set_optimizer(model::Model, optimizer_factory;
37 bridge_constraints::Bool=true)
38
39
40 Creates an empty `MathOptInterface.AbstractOptimizer` instance by calling
41 `optimizer_factory()` and sets it as the optimizer of `model`. Specifically,
42 `optimizer_factory` must be callable with zero arguments and return an empty
43 `MathOptInterface.AbstractOptimizer`.
44
45 If `bridge_constraints` is true, constraints that are not supported by the
46 optimizer are automatically bridged to equivalent supported constraints when
47 an appropriate transformation is defined in the `MathOptInterface.Bridges`
48 module or is defined in another module and is explicitly added.
49
50 See [`set_optimizer_attributes`](@ref) and [`set_optimizer_attribute`](@ref) for setting
51 solver-specific parameters of the optimizer.
52
53 ## Examples
54 ```julia
55 model = Model()
56 set_optimizer(model, GLPK.Optimizer)
57 ```
58 """
59 function set_optimizer(model::Model, optimizer_constructor;
60 bridge_constraints::Bool=true)
61 error_if_direct_mode(model, :set_optimizer)
62 if bridge_constraints
63 # We set `with_names=false` because the names are handled by the first
64 # caching optimizer. If `default_copy_to` without names is supported,
65 # no need for a second cache.
66 optimizer = MOI.instantiate(optimizer_constructor, with_bridge_type=Float64, with_names=false)
67 for bridge_type in model.bridge_types
68 _moi_add_bridge(optimizer, bridge_type)
69 end
70 else
71 optimizer = MOI.instantiate(optimizer_constructor)
72 end
73 MOIU.reset_optimizer(model, optimizer)
74 end
