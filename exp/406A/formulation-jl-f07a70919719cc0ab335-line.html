555 print(io, getname(form, constr), " : ")
556 for (varid, coeff) in getcoefmatrix(form)[constrid, :]
557 name = getname(form, varid)
558 op = (coeff < 0.0) ? "-" : "+"
559 print(io, op, " ", abs(coeff), " ", name, " ")
560 end
561 op = "<="
562 if getcursense(form, constr) == Equal
563 op = "=="
564 elseif getcursense(form, constr) == Greater
565 op = ">="
566 end
567 print(io, " ", op, " ", getcurrhs(form, constr))
568 println(io, " (", getduty(constrid), constrid, " | ", isexplicit(form, constr) ,")")
569 return
570 end

573 constrs = getconstrs(form)

575 for constr_id in ids
576 if iscuractive(form, constr_id)
577 _show_constraint(io, form, constr_id)
578 end
579 end

581 end
