168 # Dummy callback function for internal use only. Responsible for updating the
169 # objective bound, saving the mip gap, and calling the user's callback.
170 #
171 # !!! Very Important Note
172 #
173 # If Julia throws an exception from within the callback, the GLPK model does not
174 # gracefully exit! Instead, it throws a `glp_delete_prob: operation not
175 # supported` error when Julia tries to finalize `prob`. This is very annoying to
176 # debug.
177 #
178 # As a work-around, we catch all Julia exceptions with a try-catch, terminate
179 # the callback with `ios_terminate`, store the exception in `cb_data.exception`,
180 # and then re-throw it once we have gracefully exited from the callback.
181 #
182 # See also: the note in `_solve_mip_problem`.
183 function _set_callback(model::Optimizer, callback_function::Function)
184 internal_callback = (tree::Ptr{Cvoid}, info::Ptr{Cvoid}) -> begin
185 cb_data = unsafe_pointer_to_objref(info)::CallbackData
186 node = glp_ios_best_node(tree)
187 if node != 0
188 model.objective_bound = glp_ios_node_bound(tree, node)
189 model.relative_gap = glp_ios_mip_gap(tree)
190 end
191 try
192 cb_data.tree = tree
193 callback_function(cb_data)
194 catch ex
195 glp_ios_terminate(tree)
196 cb_data.exception = ex
197 end
198 return Cint(0)
199 end
200 c_callback = @cfunction(
201 $internal_callback, Cint, (Ptr{Cvoid}, Ptr{Cvoid})
202 )
203 model.callback_data = CallbackData(c_callback, C_NULL, nothing)
204 model.intopt_param.cb_func = c_callback.ptr
205 model.intopt_param.cb_info = pointer_from_objref(model.callback_data)
206 return

1053 """
1054 _add_affine_constraint(
1055 problem::Union{Optimizer, Ptr{glp_prob}},
1056 columns::Vector{Cint},
1057 coefficients::Vector{Float64},
1058 sense::Cchar,
1059 rhs::Float64,
1060 )
1061
1062 Helper function to add a row to the problem. Sense must be one of `'E'`
1063 (ax == b), `'G'` (ax >= b), `'L'` (ax <= b).
1064 """
1065 function _add_affine_constraint(
1066 problem::Union{Optimizer, Ptr{glp_prob}},
1067 indices::Vector{Cint},
1068 coefficients::Vector{Float64},
1069 sense::Cchar,
1070 rhs::Float64,
1071 )
1072 if length(indices) != length(coefficients)
1073 error("columns and coefficients have different lengths.")
1074 end
1075 glp_add_rows(problem, 1)
1076 row = glp_get_num_rows(problem)
1077 glp_set_mat_row(
1078 problem, row, length(indices), offset(indices), offset(coefficients)
1079 )
1080 # According to http://most.ccib.rutgers.edu/glpk.pdf page 22, the `lb`
1081 # argument is ignored for constraint types with no lower bound (GLP_UP) and
1082 # the `ub` argument is ignored for constraint types with no upper bound
1083 # (GLP_LO). We pass ±GLP_DBL_MAX for those unused bounds since (a) we have
1084 # to pass something, and (b) it is consistent with the other usages of
1085 # ±GLP_DBL_MAX to represent infinite bounds in the rest of the GLPK
1086 # interface.
1087 if sense == Cchar('E')
1088 glp_set_row_bnds(problem, row, GLP_FX, rhs, rhs)
1089 elseif sense == Cchar('G')
1090 glp_set_row_bnds(problem, row, GLP_LO, rhs, GLP_DBL_MAX)
1091 else
1092 @assert sense == Cchar('L')
1093 glp_set_row_bnds(problem, row, GLP_UP, -GLP_DBL_MAX, rhs)
1094 end
1095 return
1096 end

1317 """
1318 _round_bounds_to_integer(model)
1319
1320 GLPK does not allow integer variables with fractional bounds. Therefore, we
1321 round the bounds of binary and integer variables to integer values prior to
1322 solving.
1323
1324 Returns a tuple `(column, lower, upper)` for the bounds that need to be reset.
1325 """
1326 function _round_bounds_to_integer(model::Optimizer)
1327 bounds_to_reset = Tuple{Int, Float64, Float64}[]
1328 for (key, info) in model.variable_info
1329 if info.type == BINARY || info.type == INTEGER
1330 lb = glp_get_col_lb(model, info.column)
1331 ub = glp_get_col_ub(model, info.column)
1332 new_lb = ceil(lb)
1333 new_ub = floor(ub)
1334 if info.type == BINARY
1335 new_lb = max(0.0, new_lb)
1336 new_ub = min(1.0, new_ub)
1337 end
1338 if new_lb != lb || new_ub != ub
1339 push!(bounds_to_reset, (info.column, lb, ub))
1340 _set_variable_bound(model, info.column, new_lb, new_ub)
1341 end
1342 end
1343 end
1344 return bounds_to_reset
1345 end

1347 function _solve_mip_problem(model::Optimizer)
1348 bounds_to_reset = _round_bounds_to_integer(model)
1349 # Because we're muddling with the presolve in this function, cache the
1350 # original setting so that it can be reset.
1351 presolve_cache = model.intopt_param.presolve
1352 try
1353 # glp_intopt requires a starting basis for the LP relaxation. There are
1354 # two ways to get this. If presolve=GLP_ON, then the presolve will find
1355 # a basis. If presolve=GLP_OFF, then we should solve the problem via
1356 # glp_simplex first.
1357 if model.intopt_param.presolve == GLP_OFF
1358 glp_simplex(model, model.simplex_param)
1359 if glp_get_status(model) != GLP_OPT
1360 # We didn't find an optimal solution to the LP relaxation, so
1361 # let's turn presolve on and let intopt figure out what the
1362 # problem is.
1363 model.intopt_param.presolve = GLP_ON
1364 end
1365 end
1366 model.solver_status = glp_intopt(model, model.intopt_param)
1367 model.last_solved_by_mip = true
1368
1369 # !!! Very Important Note
1370 #
1371 # This next bit is _very_ important! See the note associated with
1372 # set_callback.
1373 if model.callback_data.exception !== nothing
1374 throw(model.callback_data.exception)
1375 end
1376
1377 finally
1378 for (column, lower, upper) in bounds_to_reset
1379 _set_variable_bound(model, column, lower, upper)
1380 end
1381 model.intopt_param.presolve = presolve_cache
1382 end
1383 return
1384 end

1908 function MOI.get(model::Optimizer, ::MOI.ListOfConstraints)
1909 constraints = Set{Tuple{DataType, DataType}}()
1910 for info in values(model.variable_info)
1911 if info.bound == NONE
1912 elseif info.bound == LESS_THAN
1913 push!(constraints, (MOI.SingleVariable, MOI.LessThan{Float64}))
1914 elseif info.bound == GREATER_THAN
1915 push!(constraints, (MOI.SingleVariable, MOI.GreaterThan{Float64}))
1916 elseif info.bound == LESS_AND_GREATER_THAN
1917 push!(constraints, (MOI.SingleVariable, MOI.LessThan{Float64}))
1918 push!(constraints, (MOI.SingleVariable, MOI.GreaterThan{Float64}))
1919 elseif info.bound == EQUAL_TO
1920 push!(constraints, (MOI.SingleVariable, MOI.EqualTo{Float64}))
1921 elseif info.bound == INTERVAL
1922 push!(constraints, (MOI.SingleVariable, MOI.Interval{Float64}))
1923 end
1924 if info.type == CONTINUOUS
1925 elseif info.type == BINARY
1926 push!(constraints, (MOI.SingleVariable, MOI.ZeroOne))
1927 elseif info.type == INTEGER
1928 push!(constraints, (MOI.SingleVariable, MOI.Integer))
1929 end
1930 end
1931 for info in values(model.affine_constraint_info)
1932 push!(constraints, (MOI.ScalarAffineFunction{Float64}, typeof(info.set)))
1933 end
1934 return collect(constraints)
